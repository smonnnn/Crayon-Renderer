#version 400 core

in vec3 pos;
in vec2 texUV;
in vec3 normal;

out vec2 pass_texUV;
out vec3 surfaceNormal;
out vec3 toLightVector;

uniform mat4 mat_projection;
uniform mat4[] mat_transformations;
uniform mat4 mat_view;
uniform vec3 lightPos;

void main(void){
    vec4 worldPos = mat_transformations[gl_InstanceID] * vec4(pos, 1.0);
    gl_Position = mat_projection * mat_view * worldPos;
    pass_texUV = texUV;
    surfaceNormal = (mat_transformations[gl_InstanceID] * vec4(normal, 0.0)).xyz;
    toLightVector = lightPos - worldPos.xyz;
}
